using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using Microsoft.Extensions.Logging;

namespace TempoScribePro.Application.Publishing
{
    #region Ports (Hexagon-facing interfaces)

    /// <summary>
    /// Repository abstraction used for querying and updating scheduled posts.
    /// </summary>
    public interface IScheduledPostRepository
    {
        /// <summary>
        /// Retrieves posts that are scheduled to be published on or before the supplied UTC timestamp.
        /// Only returns posts that are still in 'Scheduled' status.
        /// </summary>
        Task<IReadOnlyList<Post>> FetchDueAsync(DateTime utcNow, CancellationToken ct);

        /// <summary>
        /// Persists an updated post to the store.
        /// </summary>
        Task UpdateAsync(Post post, CancellationToken ct);
    }

    /// <summary>
    /// Port responsible for actual publication side-effects (search index, cache invalidation, etc.).
    /// </summary>
    public interface IPostPublisher
    {
        /// <summary>
        /// Executes the publish operation and returns a result that may contain
        /// warnings, URLs, or additional metadata.
        /// </summary>
        Task<PublishResult> PublishAsync(Post post, CancellationToken ct);
    }

    /// <summary>
    /// Abstraction around UTC time for deterministic testing.
    /// </summary>
    public interface IClock
    {
        DateTime UtcNow { get; }
    }

    /// <summary>
    /// Raised after a post is successfully published.
    /// </summary>
    public record PostPublishedEvent(Guid PostId, DateTime PublishedAtUtc, IReadOnlyDictionary<string, string>? Metadata);

    /// <summary>
    /// Dispatches domain events to interested adapters (e.g., WebSocket hub, SignalR, email bridge).
    /// </summary>
    public interface IDomainEventDispatcher
    {
        Task DispatchAsync<TEvent>(TEvent domainEvent, CancellationToken ct);
    }

    /// <summary>
    /// Provides a distributed mutex to ensure only one node performs scheduled publishing at a time.
    /// </summary>
    public interface IDistributedLock
    {
        /// <summary>
        /// Attempts to acquire a named lock for the specified duration.
        /// Returns <see langword="true" /> if the lock was acquired.
        /// </summary>
        Task<bool> TryAcquireAsync(string name, TimeSpan holdFor, CancellationToken ct);
    }

    #endregion

    #region Domain Model (abbreviated)

    /// <summary>
    /// Simplified domain entity representing a blog post.
    /// </summary>
    public class Post
    {
        public Guid Id { get; init; }
        public DateTime? ScheduledAtUtc { get; private set; }
        public PostStatus Status { get; private set; }
        public DateTime? PublishedAtUtc { get; private set; }

        public void MarkAsPublished(DateTime utcNow)
        {
            Status = PostStatus.Published;
            PublishedAtUtc = utcNow;
            ScheduledAtUtc = null;
        }
    }

    public enum PostStatus
    {
        Draft = 0,
        Scheduled = 1,
        Published = 2,
        Archived = 3
    }

    public record PublishResult(Uri CanonicalUrl, IReadOnlyDictionary<string, string>? Metadata);

    #endregion

    #region Service

    /// <summary>
    /// Hosted service that wakes up on a configured interval and publishes scheduled posts that are due.
    /// </summary>
    public sealed class ScheduledPublishingService
    {
        private static readonly TimeSpan LockDuration       = TimeSpan.FromMinutes(1);
        private static readonly string   LockName           = "tempo_scribe_scheduled_publish";
        private static readonly TimeSpan ScanInterval       = TimeSpan.FromMinutes(1);
        private static readonly int      BatchSize          = 50;

        private readonly IScheduledPostRepository _repository;
        private readonly IPostPublisher            _publisher;
        private readonly IDomainEventDispatcher    _dispatcher;
        private readonly IClock                    _clock;
        private readonly IDistributedLock          _lock;
        private readonly ILogger<ScheduledPublishingService> _logger;

        public ScheduledPublishingService(
            IScheduledPostRepository repository,
            IPostPublisher publisher,
            IDomainEventDispatcher dispatcher,
            IClock clock,
            IDistributedLock distributedLock,
            ILogger<ScheduledPublishingService> logger)
        {
            _repository  = repository  ?? throw new ArgumentNullException(nameof(repository));
            _publisher   = publisher   ?? throw new ArgumentNullException(nameof(publisher));
            _dispatcher  = dispatcher  ?? throw new ArgumentNullException(nameof(dispatcher));
            _clock       = clock       ?? throw new ArgumentNullException(nameof(clock));
            _lock        = distributedLock ?? throw new ArgumentNullException(nameof(distributedLock));
            _logger      = logger      ?? throw new ArgumentNullException(nameof(logger));
        }

        /// <summary>
        /// Starts the infinite polling loop. Intended to be called by a hosted service
        /// such as <see cref="Microsoft.Extensions.Hosting.BackgroundService"/> or
        /// any orchestrated worker process.
        /// </summary>
        public async Task RunAsync(CancellationToken stoppingToken)
        {
            _logger.LogInformation("ScheduledPublishingService started.");

            while (!stoppingToken.IsCancellationRequested)
            {
                try
                {
                    if (await _lock.TryAcquireAsync(LockName, LockDuration, stoppingToken).ConfigureAwait(false))
                    {
                        await PerformScanAsync(stoppingToken).ConfigureAwait(false);
                    }
                    else
                    {
                        _logger.LogDebug("Could not acquire distributed lock '{LockName}'. Another node is likely performing the scan.", LockName);
                    }
                }
                catch (Exception ex)
                {
                    _logger.LogError(ex, "Unhandled exception in ScheduledPublishingService loop.");
                }

                try
                {
                    await Task.Delay(ScanInterval, stoppingToken).ConfigureAwait(false);
                }
                catch (TaskCanceledException) { /* graceful exit */ }
            }

            _logger.LogInformation("ScheduledPublishingService stopped.");
        }

        /// <summary>
        /// Retrieves and publishes all due posts, handling them in batches for memory efficiency.
        /// </summary>
        private async Task PerformScanAsync(CancellationToken ct)
        {
            var utcNow = _clock.UtcNow;
            var scannedCount  = 0;
            var publishedCount = 0;

            while (!ct.IsCancellationRequested)
            {
                IReadOnlyList<Post> due = await _repository.FetchDueAsync(utcNow, ct).ConfigureAwait(false);
                if (due.Count == 0)
                    break;

                foreach (var post in due.Take(BatchSize))
                {
                    scannedCount++;

                    try
                    {
                        var publishResult = await _publisher.PublishAsync(post, ct).ConfigureAwait(false);

                        post.MarkAsPublished(utcNow);
                        await _repository.UpdateAsync(post, ct).ConfigureAwait(false);

                        var evt = new PostPublishedEvent(post.Id, utcNow, publishResult.Metadata);
                        await _dispatcher.DispatchAsync(evt, ct).ConfigureAwait(false);

                        publishedCount++;
                        _logger.LogInformation("Published scheduled post {PostId} at {PublishedAtUtc}.", post.Id, utcNow);
                    }
                    catch (Exception ex)
                    {
                        // Log and continue processing the rest of the batch; do not stop the loop.
                        _logger.LogError(ex, "Failed to publish scheduled post {PostId}. Post will be retried on next scan.", post.Id);
                    }
                }

                // If fewer than BatchSize returned, no more due posts remain.
                if (due.Count < BatchSize) break;
            }

            if (scannedCount > 0)
            {
                _logger.LogInformation("Scheduled publishing scan complete. Scanned: {Scanned}, Published: {Published}, UtcNow: {UtcNow}.",
                    scannedCount, publishedCount, utcNow);
            }
        }
    }

    #endregion
}
