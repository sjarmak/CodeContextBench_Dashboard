// File: src/module_27.txt
// Project: TempoScribe Pro – Hexagonal Productivity Blog Suite (web_blog)
// Language: C#
// Description: Domain-centric analytics service (Port + default implementation)
// --------------------------------------------------------------------------------------------------------------------
// NOTE: In the Hexagonal architecture, this file lives inside the Domain layer. It exposes the IEditorialAnalyticsService
//       port and a default EditorialAnalyticsService implementation that depends only on other domain ports
//       (repositories, clock, logger abstraction). Infrastructure-specific details are injected at runtime by adapters.
// --------------------------------------------------------------------------------------------------------------------
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

namespace TempoScribePro.Domain.Analytics
{
    #region Ports (Domain Interfaces)

    /// <summary>
    /// Repository abstraction for persisted <see cref="WorkSession"/> aggregates.
    /// Typical implementations query SQL or NoSQL stores; the domain cares only about contracts.
    /// </summary>
    public interface IWorkSessionRepository
    {
        Task<IReadOnlyList<WorkSession>> GetSessionsAsync(DateTime fromUtc, DateTime toUtc, CancellationToken ct = default);
    }

    /// <summary>
    /// Repository abstraction for editorial tasks (<see cref="EditorialTask"/>).
    /// </summary>
    public interface IEditorialTaskRepository
    {
        Task<IReadOnlyList<EditorialTask>> GetCompletedTasksAsync(DateTime fromUtc, DateTime toUtc, CancellationToken ct = default);
    }

    /// <summary>
    /// Repository abstraction for blog posts.
    /// </summary>
    public interface IPostRepository
    {
        Task<IReadOnlyList<Post>> GetPublishedPostsAsync(DateTime fromUtc, DateTime toUtc, CancellationToken ct = default);
    }

    /// <summary>
    /// Abstraction over a clock to facilitate deterministic testing.
    /// </summary>
    public interface ISystemClock
    {
        DateTime UtcNow { get; }
    }

    /// <summary>
    /// Logger abstraction that can be wired to Serilog, NLog, etc.
    /// </summary>
    public interface ILogger
    {
        void Debug(string message, params object[] args);
        void Info (string message, params object[] args);
        void Warn (string message, params object[] args);
        void Error(string message, Exception ex, params object[] args);
    }

    /// <summary>
    /// Main analytics port—calculates productivity and velocity KPIs for editors.
    /// </summary>
    public interface IEditorialAnalyticsService
    {
        /// <summary>
        /// Generates a productivity report using data points collected since <paramref name="fromUtc"/>.
        /// </summary>
        Task<EditorProductivityReport> GenerateReportAsync(
            DateTime fromUtc,
            CancellationToken ct = default);
    }

    #endregion

    #region Entities & Value Objects (Domain)

    /// <summary>
    /// Aggregate root representing a focused work-session by an editor.
    /// </summary>
    public sealed record WorkSession(
        Guid      Id,
        Guid      UserId,
        DateTime  StartedUtc,
        DateTime  EndedUtc,
        int       WordsTyped);

    /// <summary>
    /// Aggregate root representing an editorial Kanban task.
    /// </summary>
    public sealed record EditorialTask(
        Guid      Id,
        Guid      AssigneeId,
        string    Title,
        DateTime  CreatedUtc,
        DateTime  CompletedUtc);

    /// <summary>
    /// Aggregate root representing a post.
    /// </summary>
    public sealed record Post(
        Guid     Id,
        Guid     AuthorId,
        string   Title,
        int      WordCount,
        DateTime PublishedUtc);

    #endregion

    #region DTOs / Read Models

    /// <summary>
    /// Computed KPI metrics for editors and administrators.
    /// Immutable, serializable read-model returned by analytics service.
    /// </summary>
    public sealed record EditorProductivityReport
    {
        public DateTime   FromUtc                 { get; init; }
        public DateTime   ToUtc                   { get; init; }
        public int        TotalSessions           { get; init; }
        public TimeSpan   AverageSessionDuration  { get; init; }
        public double     AverageWordsPerSession  { get; init; }
        public int        TotalWordsTyped         { get; init; }
        public int        CompletedTasks          { get; init; }
        public TimeSpan?  MedianTaskCycleTime     { get; init; }
        public int        PostsPublished          { get; init; }
        public int        TotalWordsPublished     { get; init; }
        public double     WordsPerHour            { get; init; }
    }

    #endregion

    #region Service Implementation (Domain Layer)

    /// <summary>
    /// Default, domain-level implementation of <see cref="IEditorialAnalyticsService"/>.
    /// Relies exclusively on other domain ports; free of infrastructure concerns.
    /// </summary>
    public sealed class EditorialAnalyticsService : IEditorialAnalyticsService
    {
        private readonly IWorkSessionRepository      _sessionRepo;
        private readonly IEditorialTaskRepository    _taskRepo;
        private readonly IPostRepository             _postRepo;
        private readonly ISystemClock                _clock;
        private readonly ILogger                     _log;

        public EditorialAnalyticsService(
            IWorkSessionRepository   sessionRepo,
            IEditorialTaskRepository taskRepo,
            IPostRepository          postRepo,
            ISystemClock             clock,
            ILogger                  log)
        {
            _sessionRepo = sessionRepo  ?? throw new ArgumentNullException(nameof(sessionRepo));
            _taskRepo    = taskRepo     ?? throw new ArgumentNullException(nameof(taskRepo));
            _postRepo    = postRepo     ?? throw new ArgumentNullException(nameof(postRepo));
            _clock       = clock        ?? throw new ArgumentNullException(nameof(clock));
            _log         = log          ?? throw new ArgumentNullException(nameof(log));
        }

        /// <inheritdoc />
        public async Task<EditorProductivityReport> GenerateReportAsync(DateTime fromUtc, CancellationToken ct = default)
        {
            // Ensure the time window is valid.
            var now = _clock.UtcNow;
            Guard.Against(fromUtc > now, "Parameter fromUtc cannot be in the future.");

            // Run repository queries in parallel to shorten wait times (I/O bound).
            var sessionsTask = _sessionRepo.GetSessionsAsync(fromUtc, now, ct);
            var tasksTask    = _taskRepo   .GetCompletedTasksAsync(fromUtc, now, ct);
            var postsTask    = _postRepo   .GetPublishedPostsAsync(fromUtc, now, ct);

            await Task.WhenAll(sessionsTask, tasksTask, postsTask).ConfigureAwait(false);

            var sessions = sessionsTask.Result;
            var tasks    = tasksTask.Result;
            var posts    = postsTask.Result;

            _log.Debug("Fetched {Sessions} sessions, {Tasks} tasks, {Posts} posts.", 
                       sessions.Count, tasks.Count, posts.Count);

            // Calculate session metrics.
            var sessionDurations = sessions
                .Select(s => (s.EndedUtc - s.StartedUtc).Duration())
                .Where(d => d > TimeSpan.Zero)
                .ToList();

            var totalSessionDuration = sessionDurations.Aggregate(TimeSpan.Zero, (sum, span) => sum + span);
            var totalWordsTyped      = sessions.Sum(s => s.WordsTyped);

            // Task metrics.
            var taskCycleTimes = tasks
                .Select(t => (t.CompletedUtc - t.CreatedUtc).Duration())
                .Where(d => d > TimeSpan.Zero)
                .OrderBy(d => d)
                .ToList();

            // Post metrics.
            var totalWordsPublished = posts.Sum(p => p.WordCount);

            // Derived KPIs.
            var averageSessionDuration = sessionDurations.Count == 0
                ? TimeSpan.Zero
                : TimeSpan.FromMilliseconds(sessionDurations.Average(d => d.TotalMilliseconds));

            var averageWordsPerSession = sessions.Count == 0
                ? 0
                : totalWordsTyped / (double)sessions.Count;

            var medianTaskCycleTime = taskCycleTimes.Count == 0
                ? (TimeSpan?)null
                : taskCycleTimes[taskCycleTimes.Count / 2]; // floor division OK for 0-based index after ordering.

            var hoursWorked = totalSessionDuration.TotalHours;
            var wordsPerHour = hoursWorked <= 0
                ? 0
                : totalWordsTyped / hoursWorked;

            var report = new EditorProductivityReport
            {
                FromUtc                = fromUtc,
                ToUtc                  = now,
                TotalSessions          = sessions.Count,
                AverageSessionDuration = averageSessionDuration,
                AverageWordsPerSession = Math.Round(averageWordsPerSession, 2),
                TotalWordsTyped        = totalWordsTyped,
                CompletedTasks         = tasks.Count,
                MedianTaskCycleTime    = medianTaskCycleTime,
                PostsPublished         = posts.Count,
                TotalWordsPublished    = totalWordsPublished,
                WordsPerHour           = Math.Round(wordsPerHour, 2)
            };

            _log.Info("Generated productivity report from {FromUtc} to {ToUtc}. Sessions: {Sessions}, Tasks: {Tasks}, Posts: {Posts}.",
                      report.FromUtc, report.ToUtc, report.TotalSessions, report.CompletedTasks, report.PostsPublished);

            return report;
        }
    }

    #endregion

    #region Guard Helper

    /// <summary>
    /// Short helper class for lightweight argument validation without external dependency.
    /// </summary>
    internal static class Guard
    {
        public static void Against(bool condition, string message)
        {
            if (condition)
            {
                throw new ArgumentException(message);
            }
        }
    }

    #endregion
}
