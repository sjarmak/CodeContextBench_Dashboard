// --------------------------------------------------------------------------------------------------------------------

// <summary>
//   Central location for framework-wide constants and utility helpers. These values are
//   referenced from domain, application, and adapter layers through dependency injection
//   to avoid magic strings and hidden coupling. Although the file lives in the “common”
//   assembly, the contained types remain technology-agnostic so that the Core does not

#nullable enable
using System;
using System.Diagnostics.CodeAnalysis;
using System.Globalization;

namespace TempoScribePro.WebBlog.Common
{
    /// <summary>
    /// Global constant definitions and helper utilities used across TempoScribe Pro.
    /// </summary>
    public static class Constants
    {
        /// <summary>
        /// Application information. Useful for telemetry enrichment and branding.
        /// </summary>
        public static class Application
        {
            public const string Name           = "TempoScribe Pro – Hexagonal Productivity Blog Suite";
            public const string ShortName      = "TempoScribe";
            public const string SupportEmail   = "support@temposcribe.io";
            public const string Version        = "3.2.0"; // Updated via CI when tagging.
            public const string Culture        = "en-US";
            public static readonly CultureInfo DefaultCulture = new(Culture);

            /// <summary>
            /// Returns a combined User-Agent string used for outbound HTTP calls.
            /// Format: TempoScribe/3.2.0 (+https://temposcribe.io)
            /// </summary>
            public static string UserAgent =>
                $"{ShortName}/{Version} (+https://temposcribe.io)";
        }

        /// <summary>
        /// Well-known role names used by ASP.NET Core Identity and policy
        /// authorization. They are centralized here to prevent typos.
        /// </summary>
        public static class Roles
        {
            public const string Administrator = "Administrator";
            public const string Editor        = "Editor";
            public const string Author        = "Author";
            public const string Subscriber    = "Subscriber";
            public const string Guest         = "Guest";
        }

        /// <summary>
        /// Identity Server / JWT claim types utilised in the application.
        /// </summary>
        public static class ClaimTypes
        {
            public const string TimeZone     = "tz";      // User chosen time-zone id.
            public const string Subscription = "plan";    // Monetization tier.
            public const string FeatureFlag  = "ff";      // Compressed feature flag metadata.
        }

        /// <summary>
        /// Policy names registered in <c>Startup.ConfigureServices</c>.
        /// </summary>
        public static class Policies
        {
            public const string RequireAdmin           = nameof(RequireAdmin);
            public const string RequirePaymentGate     = nameof(RequirePaymentGate);
            public const string RequireVerifiedAccount = nameof(RequireVerifiedAccount);
            public const string RequireEditorRights    = nameof(RequireEditorRights);
        }

        /// <summary>
        /// Scoped cache keys used with distributed caching layers (e.g. Redis).
        /// Prefixes should be unique to prevent collisions with other micro-services.
        /// </summary>
        public static class CacheKeys
        {
            private const string Prefix = "ts:"; // TempoScribe prefix.

            public static string Post(Guid postId)             => $"{Prefix}post:{postId:N}";
            public static string PostPreview(Guid postId)      => $"{Prefix}post-prev:{postId:N}";
            public static string UserSettings(Guid userId)     => $"{Prefix}user-settings:{userId:N}";
            public static string EditorialKanbanBoard()        => $"{Prefix}kanban:board";
            public static string TagCloud()                    => $"{Prefix}tags:cloud";
            public static string SiteMapXml()                  => $"{Prefix}sitemap:xml";
            public static string HomePage()                    => $"{Prefix}page:home";

            /// <summary>
            /// Default sliding expiration for hot data (in minutes).
            /// </summary>
            public const int DefaultSlidingMinutes = 30;
        }

        /// <summary>
        /// Feature flags toggled at runtime—this allows for dark-launching capabilities
        /// without redeploying code. Flags can be enabled via database or remote config.
        /// </summary>
        public static class FeatureFlags
        {
            public const string AiAssistedDrafts      = "ai_assisted_drafts";
            public const string NewEditorExtension    = "wmx_editor_extension";
            public const string AdvancedAnalyticsV2   = "advanced_analytics_v2";
            public const string SocialPreviewCards    = "social_preview_cards";
            public const string KanbanAutoRefresh     = "kanban_auto_refresh";
        }

        /// <summary>
        /// Queue names consumed by background workers.
        /// </summary>
        public static class Queues
        {
            public const string OutboundEmail    = "email";
            public const string WebhookDispatch  = "webhook";
            public const string BillingPipeline  = "billing";
            public const string ImageProcessing  = "imagery";
            public const string SearchIndexer    = "search_indexer";
        }

        /// <summary>
        /// Event names emitted to the internal event-bus (e.g. MediatR, Kafka topics).
        /// </summary>
        public static class EventNames
        {
            public const string PostPublished             = "post.published";
            public const string CommentModerationRequired = "comment.moderation_required";
            public const string PaymentSucceeded          = "payment.succeeded";
            public const string SubscriptionUpgraded      = "subscription.upgraded";
            public const string HealthSnapshotRequested   = "health.snapshot_requested";
        }

        /// <summary>
        /// Common string lengths used throughout database and validation layers.
        /// </summary>
        public static class Lengths
        {
            public const int Tiny      = 32;
            public const int Short     = 128;
            public const int Medium    = 512;
            public const int Long      = 2048;
            public const int VeryLong  = 8192;

            /// <summary>
            /// Ensures the provided <paramref name="value"/> does not exceed the requested
            /// <paramref name="maximumLength"/>. Throws <see cref="ArgumentException"/> if it does.
            /// </summary>
            /// <exception cref="ArgumentException">When the string is longer than allowed.</exception>
            public static string Validate(
                string value,
                int maximumLength,
                [NotNull] string? parameterName = null)
            {
                if (value is null)
                {
                    throw new ArgumentNullException(parameterName ?? nameof(value));
                }

                if (value.Length > maximumLength)
                {
                    throw new ArgumentException(
                        $"String exceeds maximum length of {maximumLength} characters.",
                        parameterName ?? nameof(value));
                }

                return value;
            }
        }

        /// <summary>
        /// Utility helpers for working with date/time formatting that are consistent across
        /// logging, API responses, and database storage.
        /// </summary>
        public static class DateFormats
        {
            // ISO-8601 variant with milliseconds (e.g. 2023-07-10T14:23:01.123Z)
            public const string Iso8601WithMs = "yyyy'-'MM'-'dd'T'HH':'mm':'ss.fff'Z'";

            // Friendly human-readable format (e.g. Jul 10, 2023 14:23)
            public const string HumanFriendly = "MMM dd, yyyy HH:mm";

            /// <summary>
            /// Converts the <paramref name="utcDate"/> to ISO-8601 string with milliseconds precision.
            /// </summary>
            public static string ToIso8601Utc(DateTime utcDate)
            {
                if (utcDate.Kind != DateTimeKind.Utc)
                {
                    utcDate = utcDate.ToUniversalTime();
                }

                return utcDate.ToString(Iso8601WithMs, CultureInfo.InvariantCulture);
            }
        }
    }
}
