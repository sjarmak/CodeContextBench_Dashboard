```rust
//! src/constants.rs
//!
//! Global, application-wide constants for the _VisuTility Orchestrator_ project.
//!
//! This module centralises all “magic numbers”, environment-variable names,
//! default network ports, timeouts, and other immutable settings that are
//! consumed across multiple layers of the code-base.  Co-locating them here
//! helps to keep the rest of the implementation tidy and makes it trivial to
//! discover and change process-wide parameters.
//!
//! ## Usage
//!
//! ```rust
//! use visutility_orchestrator::constants::*;
//!
//! let addr = *DEFAULT_HTTP_SOCKET_ADDR;       // 0.0.0.0:7812
//! let ns   = MODEL_REGISTRY_NAMESPACE;        // "visutility"
//! ```
//!
//! ## Conventions
//!
//! 1. Immutable, compile-time literals are uppercase `const`s.
//! 2. Values that must be computed at startup (e.g. Regex, Duration) are
//!    exposed as `Lazy` statics (`once_cell`).
//! 3. All string constants meant for environment variables are suffixed with
//!    `_ENV` (e.g. `DATABASE_URL_ENV`).

use once_cell::sync::Lazy;
use regex::Regex;
use std::{
    net::SocketAddr,
    time::Duration,
};

/* -------------------------------------------------------------------------- */
/*                               Build Metadata                               */
/* -------------------------------------------------------------------------- */

/// Crate semantic version, injected by Cargo at compile-time.
pub const BUILD_SEMVER: &str = env!("CARGO_PKG_VERSION");

/// Optional short Git commit SHA (set via `RUSTFLAGS` or build-script).
pub const BUILD_GIT_SHA: &str = option_env!("GIT_SHA").unwrap_or("unknown");

/// UTC UNIX timestamp captured at compile-time (`build.rs` recommended).
pub const BUILD_TIMESTAMP_UTC: &str = option_env!("BUILD_TS_UTC").unwrap_or("unknown");

/* -------------------------------------------------------------------------- */
/*                                I/O Settings                                */
/* -------------------------------------------------------------------------- */

/// Default hostname the public HTTP server binds to.
pub const DEFAULT_HTTP_HOST: &str = "0.0.0.0";

/// Default TCP port for the public HTTP server.
pub const DEFAULT_HTTP_PORT: u16 = 7_812;

/// Socket-address combining [`DEFAULT_HTTP_HOST`] and [`DEFAULT_HTTP_PORT`].
pub static DEFAULT_HTTP_SOCKET_ADDR: Lazy<SocketAddr> = Lazy::new(|| {
    format!("{DEFAULT_HTTP_HOST}:{DEFAULT_HTTP_PORT}")
        .parse()
        .expect("failed to parse DEFAULT_HTTP_SOCKET_ADDR")
});

/// gRPC server port (kept separate from the REST/HTTP service).
pub const DEFAULT_GRPC_PORT: u16 = 9_812;

/* -------------------------------------------------------------------------- */
/*                           Environment-Variable Keys                        */
/* -------------------------------------------------------------------------- */

/// Location of the Postgres / CockroachDB instance used for metadata.
pub const DATABASE_URL_ENV: &str = "VISU_DB_URL";

/// Path on local disk or S3/GCS bucket used for artefact storage.
pub const ARTEFACT_STORE_URI_ENV: &str = "VISU_ARTEFACT_STORE_URI";

/// When present, disables all TLS enforcement (**do not** use in production).
pub const INSECURE_DEV_MODE_ENV: &str = "VISU_INSECURE_DEV_MODE";

/* -------------------------------------------------------------------------- */
/*                              Model Registry                                */
/* -------------------------------------------------------------------------- */

/// Logical namespace for all models that belong to this service instance.
pub const MODEL_REGISTRY_NAMESPACE: &str = "visutility";

/// Regular expression guaranteeing valid registry IDs:  
/// `<namespace>/<model_name>:<semver>`
pub static MODEL_REGISTRY_ID_RE: Lazy<Regex> = Lazy::new(|| {
    Regex::new(r"^[a-z0-9_\-]+/[a-z0-9_\-]+:[0-9]+\.[0-9]+\.[0-9]+$")
        .expect("invalid MODEL_REGISTRY_ID_RE regex")
});

/* -------------------------------------------------------------------------- */
/*                               Time Constants                               */
/* -------------------------------------------------------------------------- */

/// Hard deadline for pulling a single frame from a camera stream.
pub static INGEST_FRAME_DEADLINE: Lazy<Duration> =
    Lazy::new(|| Duration::from_millis(350));

/// Global timeout for completing an entire batch-inference job.
pub static BATCH_INFERENCE_TIMEOUT: Lazy<Duration> =
    Lazy::new(|| Duration::from_secs(300));

/* -------------------------------------------------------------------------- */
/*                              File-System Paths                             */
/* -------------------------------------------------------------------------- */

/// Default directory for temporary, intermediate artefacts.
pub const DEFAULT_TMP_DIR: &str = "/tmp/visu";

/// Name of the sub-directory inside [`DEFAULT_TMP_DIR`] where raw frames live.
pub const TMP_RAW_FRAMES_DIR: &str = "raw_frames";

/// Name of the sub-directory inside [`DEFAULT_TMP_DIR`] for encoded outputs.
pub const TMP_ENCODED_FRAMES_DIR: &str = "encoded";

/* -------------------------------------------------------------------------- */
/*                                  Logging                                   */
/* -------------------------------------------------------------------------- */

/// Default log level when `RUST_LOG` is not set.
pub const DEFAULT_LOG_LEVEL: &str = "info";

/// Maximum number of background log-shipping tasks.
pub const LOG_SHIPPING_TASKS: usize = 4;

/* -------------------------------------------------------------------------- */
/*                         Observability / Metrics                            */
/* -------------------------------------------------------------------------- */

/// Port where the Prometheus exporter listens.
pub const PROMETHEUS_PORT: u16 = 9_424;

/// Prometheus scrape endpoint path.
pub const PROMETHEUS_METRICS_PATH: &str = "/metrics";

/* -------------------------------------------------------------------------- */
/*                                 Features                                   */
/* -------------------------------------------------------------------------- */

/// Feature flags used throughout the codebase.
pub mod features {
    /// Enables experimental, on-device quantisation for Edge TPU deployments.
    pub const EDGE_TPU_QUANTISATION: &str = "edge_tpu_quant";
    /// Activates the auto-retraining _Observer_ pipeline.
    pub const AUTO_RETRAINING: &str = "auto_retraining";
    /// Enables hyperparameter tuning orchestrated by the _Strategy_ layer.
    pub const HYPERPARAM_TUNING: &str = "hparam_tuning";
}

/* -------------------------------------------------------------------------- */
/*                             Utility Functions                              */
/* -------------------------------------------------------------------------- */

/// Builds a fully-qualified model registry ID.
///
/// # Example
/// ```rust
/// use visutility_orchestrator::constants::build_registry_id;
///
/// let id = build_registry_id("object_detector", "1.4.2");
/// assert_eq!(id, "visutility/object_detector:1.4.2");
/// ```
pub fn build_registry_id(model_name: &str, semver: &str) -> String {
    format!("{MODEL_REGISTRY_NAMESPACE}/{model_name}:{semver}")
}

/// Returns `true` if *EDGE TPU quantisation* is enabled via Cargo features.
#[inline]
pub fn is_edge_tpu_quant_enabled() -> bool {
    cfg!(feature = "edge_tpu_quant")
}
